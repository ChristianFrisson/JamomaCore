###########################################################################
CUE Initialisation	# (fold)
###########################################################################

	# Module /output~
		/output~/alias/remove
		/output~/audio/active 0
		/output~/audio/gain 0.000
		/output~/balance 0.000
		/output~/ch.1 1
		/output~/ch.2 2
		/output~/limiter/active 0
		/output~/limiter/dcblocker/active 1
		/output~/limiter/lookahead 64
		/output~/limiter/mode exponential
		/output~/limiter/postamp 0.000
		/output~/limiter/preamp 6.000
		/output~/limiter/release 300.000
		/output~/limiter/threshold -1.000
		/output~/record/filetype aiff
		/output~/record/samptype int24
		/output~/saturation/active 1
		/output~/saturation/depth 10.000
		/output~/view/freeze 0
		/output~/view/highlight 0
	# Module /stereoPlayer
		/stereoPlayer/alias/remove
		/stereoPlayer/nextFile/loop/start 0.000
		/stereoPlayer/nextFile/loop/end 0.000
		/stereoPlayer/nextFile/rate 1.000
		/stereoPlayer/audio/gain 0.000
		/stereoPlayer/audio/mute 0
		/stereoPlayer/loop 1
		/stereoPlayer/play 0
		/stereoPlayer/view/freeze 0
		/stereoPlayer/view/highlight 0

# (end)

###########################################################################
CUE AudioOn	# (fold)
###########################################################################

	# Module /stereoPlayer
		/stereoPlayer/audio/gain 100.000

	# Module /output~
		/output~/audio/active 1
		/output~/audio/gain 75.000 ramp 2000

# (end)

###########################################################################
CUE PreloadDrums	# (fold)
###########################################################################

	# First we set the file to play, start and end positions, and playback rate
	/stereoPlayer/nextFile/open drumLoop.aif
	/stereoPlayer/nextFile/loop/start 0.
	/stereoPlayer/nextFile/loop/end 0.		# 0: Play to end of file
	/stereoPlayer/nextFile/rate 1.
	
	# Next we preload it.
	# This way it can be triggered instantly by next cue, without delays caused by disk seek times
	/stereoPlayer/nextFile/preload

# (end)

###########################################################################
CUE PlayDrums	# (fold)
###########################################################################

	# We make the preloaded file the current one
	/stereoPlayer/nextFile/makeCurrent
	
	/stereoPlayer/play 1

# (end)

###########################################################################
CUE PreloadVibes	# (fold)
###########################################################################

	# First we set the file to play, start and end positions, and playback rate
	/stereoPlayer/nextFile/open vibes-a1.aif
	/stereoPlayer/nextFile/loop/start 450	# Start playback 200 ms into the sound file
	/stereoPlayer/nextFile/loop/end 0		# Loop end point at end of sound file
	/stereoPlayer/nextFile/rate -0.5		# Reverse playback at reduced speed
	
	# Next we preload it.
	# This way it can be triggered instantly by next cue, without delays caused by disk seek times
	/stereoPlayer/nextFile/preload

# (end)

###########################################################################
CUE PlayVibes	# (fold)
###########################################################################

	# We make the preloaded file the current one
	/stereoPlayer/nextFile/makeCurrent

	/stereoPlayer/play 1
	
# (end)
